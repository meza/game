

	public void buildMainPage()
	{
		addComponentsToPane(mainFrame.getContentPane());
		mainFrame.validate();
	}

	public void addComponentsToPane(Container pane)
	{
		JPanel emptyPanel;
		JButton emptyButton = new JButton();
		JButton newGameButton = new JButton("New game");
		JButton loadGameButton = new JButton("Load game");
		Font buttonFont = new Font("Monospaced", Font.BOLD, 25);
		GridBagConstraints newGameButtonConstraints = new GridBagConstraints();
		GridBagConstraints loadGameButtonConstraints = new GridBagConstraints();
		GridBagConstraints emptyPanelConstraints = new GridBagConstraints();

		pane.setLayout(new GridBagLayout());

		//format the buttons
		newGameButton.setBackground(Color.darkGray);
		newGameButton.setForeground(Color.white);
		newGameButton.setFont(buttonFont);
		newGameButton.setBorder(border);
		newGameButton.setPreferredSize(new Dimension(200, 50));

		loadGameButton.setBackground(Color.darkGray);
		loadGameButton.setForeground(Color.white);
		loadGameButton.setFont(buttonFont);
		loadGameButton.setBorder(border);
		loadGameButton.setPreferredSize(new Dimension(200, 50));

		emptyButton.setBackground(Color.darkGray);
		emptyButton.setForeground(Color.white);
		emptyButton.setFont(buttonFont);
		emptyButton.setPreferredSize(new Dimension(200, 50));

		for (int x = 0; x < 10; x++)
		{
			for (int y = 0; y < 10; y++)
			{

				if (x == 5 && y == 5)
				{
					newGameButtonConstraints.fill = GridBagConstraints.HORIZONTAL;
					newGameButtonConstraints.gridx = x;
					newGameButtonConstraints.gridy = y;
					pane.add(newGameButton, newGameButtonConstraints);
				}
				if (x == 5 && y == 6)
				{
					loadGameButtonConstraints.fill = GridBagConstraints.HORIZONTAL;
					loadGameButtonConstraints.gridx = x;
					loadGameButtonConstraints.gridy = y;
					pane.add(loadGameButton, loadGameButtonConstraints);
				}
				else
				{
					emptyPanel = new JPanel();
					emptyPanel.setBackground(Color.darkGray);
					emptyPanel.add(emptyButton);
					String textX = Integer.toString(x);
					String textY = Integer.toString(y);
					String text = textX + "; " + textY;
					emptyButton.setText(text);
					emptyPanelConstraints.fill = GridBagConstraints.HORIZONTAL;
					emptyPanelConstraints.gridx = x;
					emptyPanelConstraints.gridy = y;
					pane.add(emptyPanel, emptyPanelConstraints);

					try
					{
						Thread.sleep(200);
					}
					catch (Exception e)
					{
					}
				}
			}
		}

		//add action listeners to the buttons
		newGameButton.addActionListener(new NewGameListener());
		loadGameButton.addActionListener(new LoadGameListener());



	}
